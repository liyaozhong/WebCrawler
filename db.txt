CREATE TABLE `moviedownloadlinks` (
	`MOVIE_NAME` VARCHAR(100) NULL DEFAULT '',
	`DOWN_LOAD_LINK` VARCHAR(1000) NULL DEFAULT ''
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;

CREATE TABLE `movieinfo` (
	`MOVIE_NAME` VARCHAR(100) NOT NULL DEFAULT '',
	`HAIBAO_PATH` VARCHAR(200) NULL DEFAULT '',
	PRIMARY KEY (`MOVIE_NAME`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;

CREATE TABLE `movienames` (
	`MOVIE_NAME` VARCHAR(100) NULL DEFAULT '',
	`MOVIE_OTHER_NAME` VARCHAR(300) NULL DEFAULT ''
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
